Module TFF.Term

type _ binder =
  1. | Exists : {
    1. type_vars : 'ty_var list;
    2. term_vars : 'term_var list;
    3. triggers : 'term list;
    } -> < ty_var : 'ty_var ; term_var : 'term_var ; term : 'term.. > binder
    (*

    Existancial quantification

    *)
  2. | Forall : {
    1. type_vars : 'ty_var list;
    2. term_vars : 'term_var list;
    3. triggers : 'term list;
    } -> < ty_var : 'ty_var ; term_var : 'term_var ; term : 'term.. > binder
    (*

    Universal quantification

    *)
  3. | Letin : ('term_var * 'term) list -> < term_var : 'term_var ; term : 'term.. > binder
    (*

    Sequential Let bindings

    *)
  4. | Letand : ('term_var * 'term) list -> < term_var : 'term_var ; term : 'term.. > binder
    (*

    Parallel Let bindings

    *)
  5. | Map : 'term_var list -> < term_var : 'term_var.. > binder
    (*

    Maps: encoding of higher-order functions in first-order.

    *)

First-order binders that can occur in terms.

type _ pattern =
  1. | Var : 'term_var -> < term_var : 'term_var.. > pattern
  2. | Constructor : 'term_cst * 'term_var list -> < term_var : 'term_var ; term_cst : 'term_cst.. > pattern

First-order patterns.

type _ view =
  1. | Var : 'term_var -> < term_var : 'term_var.. > view
    (*

    Term Variables

    *)
  2. | App : 'term_cst * 'ty list * 'term list -> < ty : 'ty ; term : 'term ; term_cst : 'term_cst ; builtin : 'blt.. > view
    (*

    Polymorphic application of a function symbol, with explicit type arguments

    *)
  3. | Match : 'term * ('t pattern * 'term) list -> (< term_var : 'term_var ; term_cst : 'term_cst ; term : 'term.. > as 't) view
    (*

    Pattern matching.

    *)
  4. | Binder : < ty_var : 'ty_var ; term_var : 'term_var ; term : 'term.. > binder * 'term -> < ty_var : 'ty_var ; term_var : 'term_var ; term : 'term.. > view
    (*

    Binders over a body term.

    *)
